## **思路**

容易想到，把辣度值升序后， p[i]取多少次是等于c[i-1][k-1]的  例如:5个取3个   **要使每次必取最后一个** 得到的就是从2个里取一个，所以不难得出计算式;

</br>

##  **代码**
``` c++
#include <bits/stdc++.h>
using namespace std;

#define MAX_N 100010
#define MAX_K 60
#define MOD 1000000007
#define long long long

long C[MAX_N][MAX_K];
long p[MAX_N];

void init(int n)
{
    for(int i = 0; i<=n; i++)
    {
        C[i][0] = 1;
        if(i < MAX_K)
            C[i][i] = 1;
        for(int j = 1; j<i && j < MAX_K; j++)
        {
            C[i][j] = (C[i-1][j] + C[i-1][j-1] ) % MOD;
        }
    }
}


int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int n, k;
    long ans = 0;
    cin >> n >> k;
    init(n);
    for(int i = 1; i<=n; i++)
        cin >> p[i];
    sort(p + 1, p+n+1);
    for(int i = k; i <= n; i++)
    {
        ans += (C[i-1][k-1] * p[i]) % MOD;
        ans %= MOD;
    }
    cout << ans;
    return 0;
}
```
