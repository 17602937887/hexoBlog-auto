## **思路**
题意很好理解，用二项式定理处理出组合数，带入计算即可；
注意不要溢出；
</br>

##  **代码**
``` c++

#include <bits/stdc++.h>
using namespace std;

#define long long long
#define MAX 1010
#define MOD 10007

long C[MAX][MAX];

void init(int n)
{
    C[0][0] = 1;
    for(int i = 1; i<=n; i++)
    {
        C[i][0] = C[i][i] = 1;
        for(int j = 1; j < i; j++)
        {
            C[i][j] = (C[i-1][j-1] + C[i-1][j]) % MOD;
        }
    }
}

long pow_mod(int n, int m)
{
    long tmp = 1;
    while(m)
    {
        if(m & 1)
            tmp = ((tmp % MOD) * (n % MOD) ) % MOD;
        n = ((n%MOD) * (n%MOD) ) % MOD;
        m >>= 1;
    }
    return tmp;
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int p, q, k, a, b;
    cin >> p >> q >> k >> a >> b;
    init(k);
    cout << pow_mod(p, a) % MOD * pow_mod(q, b) % MOD * C[k][b] % MOD;
    return 0;
}
```
