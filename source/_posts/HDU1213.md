## 题目描述：
有n个人，通过合并操作，问最后有多少个不同的集合;

<escape><!-- more --></escape>

</br>
## 思路:
并查集的基操， 用set维护一下已经出现的祖宗节点，当出现的祖宗没有在set中保存，则其为一个新的集合，ans++即可。最后输出ans即可！
</br>
## 代码:
``` C++
/*************************************************************************
	> File Name: main.cpp
	> Author: Hang_cc
	> Mail: 545192053.com 
	> Created Time: Fri 28 Sep 2018 02:33:52 PM PDT
 ************************************************************************/

#include<bits/stdc++.h>
using namespace std;

int pre[1010];

void init(int n)
{
	for(int i = 1; i<=n; i++)
		pre[i] = i;
}

int Find(int x)
{
	return x == pre[x] ? x : pre[x] = Find(pre[x]);
}

void Union(int x, int y)
{
	x = Find(x);
	y = Find(y);
	if(x != y)
	{
		pre[x] = y;
	}
}

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	int T;
	cin >> T;
	while(T--)
	{
		int ans = 0;
		int n, m;
		cin >> n >> m;
		init(n);
		for(int i = 1; i<=m; i++)
		{
			int x, y;
			cin >> x >> y;
			Union(x, y);
		}
		set<int> S;
		for(int i = 1; i<=n; i++)
		{
			if(S.find(Find(i)) == S.end())
			{
				//cout<<"i = "<<i<< "pre = "<<pre[i]<<endl;
				ans++;
				S.insert(pre[i]);
			}
		}
		cout << ans << endl;
	}
	return 0;
}
```
